22/** \brief VMU framebuffer.
23
24 This object contains a 48x32 monochrome framebuffer. It can be painted to,
25 or displayed one the VMUs connected to the system, using the API below.
26
27 \headerfile dc/vmu_fb.h
28 */
29typedef struct {
30 uint32_t data[48];
31} vmufb_t;

33/** \brief VMU framebuffer font meta-data.

44/** \brief Render into the VMU framebuffer
45
46 This function will paint the provided pixel data into the VMU framebuffer,
47 into the rectangle provided by the x, y, w and h values.
48
49 \param fb A pointer to the vmufb_t to paint to.
50 \param x The horizontal position of the top-left corner of
51 the drawing area, in pixels
52 \param y The vertical position of the top-left corner of the
53 drawing area, in pixels
54 \param w The width of the drawing area, in pixels
55 \param h The height of the drawing area, in pixels
56 \param data A pointer to the pixel data that will be painted
57 into the drawing area.
58 */
59void vmufb_paint_area(vmufb_t *fb,
60 unsigned int x, unsigned int y,
61 unsigned int w, unsigned int h,
62 const char *data);

64/** \brief Clear a specific area of the VMU framebuffer

69 \param fb A pointer to the vmufb_t to paint to.

81/** \brief Clear the VMU framebuffer

99/** \brief Render a string into the VMU framebuffer
100
101 This function uses the provided font to render text into the VMU
102 framebuffer.
103
104 \param fb A pointer to the vmufb_t to paint to.
105 \param font A pointer to the vmufb_font_t that will be used for
106 painting the text
107 \param x The horizontal position of the top-left corner of
108 the drawing area, in pixels
109 \param y The vertical position of the top-left corner of the
110 drawing area, in pixels
111 \param w The width of the drawing area, in pixels
112 \param h The height of the drawing area, in pixels
113 \param line_spacing Specify the number of empty lines that should
114 separate two lines of text
115 \param str The text to render
116 */
117void vmufb_print_string_into(vmufb_t *fb,
118 const vmufb_font_t *font,
119 unsigned int x, unsigned int y,
120 unsigned int w, unsigned int h,
121 unsigned int line_spacing,
122 const char *str);
123
124/** \brief Render a string into the VMU framebuffer
125
126 Simplified version of vmufb_print_string_into(). This is the same as calling
127 vmufb_print_string_into with x=0, y=0, w=48, h=32, line_spacing=0.
128
129 \param fb A pointer to the vmufb_t to paint to.
130 \param font A pointer to the vmufb_font_t that will be used for
131 painting the text
132 \param str The text to render
133 */
134static __inline__ void
135vmufb_print_string(vmufb_t *fb, const vmufb_font_t *font, const char *str) {
136 vmufb_print_string_into(fb, font, 0, 0, 48, 32, 0, str);
137}
