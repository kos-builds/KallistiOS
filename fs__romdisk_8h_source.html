8/** \file kos/fs_romdisk.h
9 \brief ROMFS virtual file system.
10 \ingroup vfs_romdisk
11
12 This file contains support for the romdisk VFS. This VFS allows you to make
13 Linux-style ROMFS images and either embed them into your binary or load them
14 at runtime from some other source (such as a CD-ROM). These images are made
15 with the genromfs program that is included in the utils portion of the tree.
16
17 You can choose to automount one ROMFS image by embedding it into your binary
18 and using the appropriate KOS_INIT_FLAGS() setting. The embedded ROMFS will
19 mount itself on /rd. You can also mount additional images that you load
20 from some other source on whatever mountpoint you want.
21
22 \author Megan Potter
23*/
24

53/** \brief Mount a ROMFS image as a new filesystem.
54
55 This function will mount a ROMFS image that has been loaded into memory to
56 the specified mountpoint.
57
58 \param mountpoint The directory to mount this romdisk on
59 \param img The ROMFS image
60 \param own_buffer If 0, you are still responsible for img, and must
61 free it if appropriate. If non-zero, img will be
62 freed when it is unmounted
63 \retval 0 On success
64 \retval -1 If fs_romdisk_init not called
65 \retval -2 If img is invalid
66 \retval -3 If a malloc fails
67*/
68int fs_romdisk_mount(const char * mountpoint, const uint8 *img, int own_buffer);
69
70/** \brief Unmount a ROMFS image.
71
72 This function unmounts a ROMFS image that has been previously mounted with
73 fs_romdisk_mount(). This function does not check for open files on the fs,
74 so make sure that all files have been closed before calling it. If the VFS
75 owns the buffer (own_buffer was non-zero when you called the mount function)
76 then this function will also free the buffer.
77
78 \param mountpoint The ROMFS to unmount
79 \retval 0 On success
80 \retval -1 On error
81
82 \par Error Conditions:
83 \em ENOENT - no such ROMFS was mounted
84*/
85int fs_romdisk_unmount(const char * mountpoint);
