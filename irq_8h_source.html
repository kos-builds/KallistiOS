35/** \brief The number of bytes required to save thread context.
36 \ingroup irqs
37
38 This should include all general CPU registers, FP registers, and status regs
39 (even if not all of these are actually used).
40
41 On the Dreamcast, we need 228 bytes for all of that, but we round it up to a
42 nicer number for sanity.
43*/
44#define REG_BYTE_CNT 256 /* Currently really 228 */
45
46/** \brief Architecture-specific structure for holding the processor state.
47 \ingroup irqs
48
49 This structure should hold register values and other important parts of the
50 processor state. The size of this structure should be less than or equal
51 to the REG_BYTE_CNT value.
52
53 \headerfile arch/irq.h
54*/
55typedef struct irq_context {
56 uint32_t pc; /**< \brief Program counter */
57 uint32_t pr; /**< \brief Procedure register (aka return address) */
58 uint32_t gbr; /**< \brief Global base register */
59 uint32_t vbr; /**< \brief Vector base register */
60 uint32_t mach; /**< \brief Multiply-and-accumulate register (high) */
61 uint32_t macl; /**< \brief Multiply-and-accumulate register (low) */
62 uint32_t sr; /**< \brief Status register */
63 uint32_t fpul; /**< \brief Floatint-point communication register */
64 uint32_t fr[16]; /**< \brief Primary floating point registers */
65 uint32_t frbank[16]; /**< \brief Secondary floating point registers */
66 uint32_t r[16]; /**< \brief 16 general purpose (integer) registers */
67 uint32_t fpscr; /**< \brief Floating-point status/control register */
68} irq_context_t __attribute__((aligned(32)));

107/** \defgroup irq_exception_codes Exception Codes

119/** \defgroup irq_reset_codes Reset Type
120 \brief IRQ reset type codes
121
122 These are exceptions that essentially cause a reset of the system. They
123 cannot actually be caught by normal means. They will all automatically cause
124 a branch to address 0xA0000000. These are pretty much fatal.
125
126 @{
127*/
128#define EXC_RESET_POWERON 0x0000 /**< \brief Power-on reset */
129#define EXC_RESET_MANUAL 0x0020 /**< \brief Manual reset */
130#define EXC_RESET_UDI 0x0000 /**< \brief Hitachi UDI reset */
131#define EXC_ITLB_MULTIPLE 0x0140 /**< \brief Instruction TLB multiple hit */
132#define EXC_DTLB_MULTIPLE 0x0140 /**< \brief Data TLB multiple hit */
133/** @} */
134
135/** \defgroup irq_reexec_codes Re-Execution Type
136 \brief IRQ re-execution type codes
137
138 These exceptions will stop the currently processing instruction, and
139 transition into exception processing. After handling the exception (assuming
140 that it can be handled by the code), the offending instruction will be
141 re-executed from the start.
142
143 @{
144*/
145#define EXC_USER_BREAK_PRE 0x01e0 /**< \brief User break before instruction */
146#define EXC_INSTR_ADDRESS 0x00e0 /**< \brief Instruction address */
147#define EXC_ITLB_MISS 0x0040 /**< \brief Instruction TLB miss */
148#define EXC_ITLB_PV 0x00a0 /**< \brief Instruction TLB protection violation */
149#define EXC_ILLEGAL_INSTR 0x0180 /**< \brief Illegal instruction */
150#define EXC_SLOT_ILLEGAL_INSTR 0x01a0 /**< \brief Slot illegal instruction */
151#define EXC_GENERAL_FPU 0x0800 /**< \brief General FPU exception */
152#define EXC_SLOT_FPU 0x0820 /**< \brief Slot FPU exception */
153#define EXC_DATA_ADDRESS_READ 0x00e0 /**< \brief Data address (read) */
154#define EXC_DATA_ADDRESS_WRITE 0x0100 /**< \brief Data address (write) */
155#define EXC_DTLB_MISS_READ 0x0040 /**< \brief Data TLB miss (read) */
156#define EXC_DTLB_MISS_WRITE 0x0060 /**< \brief Data TLB miss (write) */
157#define EXC_DTLB_PV_READ 0x00a0 /**< \brief Data TLB protection violation (read) */
158#define EXC_DTLB_PV_WRITE 0x00c0 /**< \brief Data TLB protection violation (write) */
159#define EXC_FPU 0x0120 /**< \brief FPU exception */
160#define EXC_INITIAL_PAGE_WRITE 0x0080 /**< \brief Initial page write exception */
161/** @} */

163/** \defgroup irq_completion_codes Completion Type
164 \brief IRQ completion type codes
165
166 These exceptions are actually handled in-between instructions, allowing the
167 instruction that causes them to finish completely. The saved PC thus is the
168 value of the next instruction.
169
170 @{
171*/
172#define EXC_TRAPA 0x0160 /**< \brief Unconditional trap (trapa) */
173#define EXC_USER_BREAK_POST 0x01e0 /**< \brief User break after instruction */
174/** @} */
